use cookDB;

	from buytbl
    group by userid
    order by userid desc;


# 문제 3
use cookDB;
select userid as '사용자ID', avg(amount) as '평균구매개수'
	from buytbl
    group by userid
    having avg(amount) >= 2
    order by avg(amount) desc;
